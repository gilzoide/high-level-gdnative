242 void hgdn_gdnative_init(const godot_gdnative_init_options *options) {
243  hgdn_core_api = options->api_struct;
244  for (const godot_gdnative_api_struct *ext = hgdn_core_api->next; ext; ext = ext->next) {
245  if (ext->version.major == 1 && ext->version.minor == 1) {
246  hgdn_core_1_1_api = (const godot_gdnative_core_1_1_api_struct *) ext;
247  } else if (ext->version.major == 1 && ext->version.minor == 2) {
248  hgdn_core_1_2_api = (const godot_gdnative_core_1_2_api_struct *) ext;
249  }
250  }
251 
252  for (unsigned int i = 0; i < hgdn_core_api->num_extensions; i++) {
253  switch(hgdn_core_api->extensions[i]->type) {
254  case GDNATIVE_EXT_NATIVESCRIPT:
255  hgdn_nativescript_api = (const godot_gdnative_ext_nativescript_api_struct *) hgdn_core_api->extensions[i];
256  for (const godot_gdnative_api_struct *ext = hgdn_nativescript_api->next; ext; ext = ext->next) {
257  if (ext->version.major == 1 && ext->version.minor == 1) {
258  hgdn_nativescript_1_1_api = (const godot_gdnative_ext_nativescript_1_1_api_struct *) ext;
259  }
260  }
261  break;
262 
263  case GDNATIVE_EXT_PLUGINSCRIPT:
264  hgdn_pluginscript_api = (const godot_gdnative_ext_pluginscript_api_struct *) hgdn_core_api->extensions[i];
265  break;
266 
267  case GDNATIVE_EXT_ANDROID:
268  hgdn_android_api = (const godot_gdnative_ext_android_api_struct *) hgdn_core_api->extensions[i];
269  break;
270 
271  case GDNATIVE_EXT_ARVR:
272  hgdn_arvr_api = (const godot_gdnative_ext_arvr_api_struct *) hgdn_core_api->extensions[i];
273  for (const godot_gdnative_api_struct *ext = hgdn_arvr_api->next; ext; ext = ext->next) {
274  if (ext->version.major == 1 && ext->version.minor == 2) {
275  hgdn_arvr_1_2_api = (const godot_gdnative_ext_arvr_1_2_api_struct *) ext;
276  }
277  }
278  break;
279 
280  case GDNATIVE_EXT_VIDEODECODER:
281  hgdn_videodecoder_api = (const godot_gdnative_ext_videodecoder_api_struct *) hgdn_core_api->extensions[i];
282  break;
283 
284  case GDNATIVE_EXT_NET:
285  hgdn_net_api = (const godot_gdnative_ext_net_api_struct *) hgdn_core_api->extensions[i];
286  for (const godot_gdnative_api_struct *ext = hgdn_net_api->next; ext; ext = ext->next) {

290  }
291  break;
292 
293  default:
294  break;
295  }
