199 void hgdn_gdnative_init(const godot_gdnative_init_options *options) {
200  hgdn_core_api = options->api_struct;
201  for (const godot_gdnative_api_struct *ext = hgdn_core_api->next; ext; ext = ext->next) {
202  if (ext->version.major == 1 && ext->version.minor == 1) {
203  hgdn_core_1_1_api = (const godot_gdnative_core_1_1_api_struct *) ext;
204  } else if (ext->version.major == 1 && ext->version.minor == 2) {
205  hgdn_core_1_2_api = (const godot_gdnative_core_1_2_api_struct *) ext;
206  }
207  }
208 
209  for(unsigned int i = 0; i < hgdn_core_api->num_extensions; i++) {
210  switch(hgdn_core_api->extensions[i]->type) {
211  case GDNATIVE_EXT_NATIVESCRIPT:
212  hgdn_nativescript_api = (const godot_gdnative_ext_nativescript_api_struct *) hgdn_core_api->extensions[i];
213  for (const godot_gdnative_api_struct *ext = hgdn_nativescript_api->next; ext; ext = ext->next) {
214  if (ext->version.major == 1 && ext->version.minor == 1) {
215  hgdn_nativescript_1_1_api = (const godot_gdnative_ext_nativescript_1_1_api_struct *) ext;
216  }
217  }
218  break;
219 
220  case GDNATIVE_EXT_PLUGINSCRIPT:
221  hgdn_pluginscript_api = (const godot_gdnative_ext_pluginscript_api_struct *) hgdn_core_api->extensions[i];
222  break;
223 
224  case GDNATIVE_EXT_ANDROID:
225  hgdn_android_api = (const godot_gdnative_ext_android_api_struct *) hgdn_core_api->extensions[i];
226  break;
227 
228  case GDNATIVE_EXT_ARVR:
229  hgdn_arvr_api = (const godot_gdnative_ext_arvr_api_struct *) hgdn_core_api->extensions[i];
230  for (const godot_gdnative_api_struct *ext = hgdn_arvr_api->next; ext; ext = ext->next) {
231  if (ext->version.major == 1 && ext->version.minor == 2) {
232  hgdn_arvr_1_2_api = (const godot_gdnative_ext_arvr_1_2_api_struct *) ext;
233  }
234  }
235  break;
236 
237  case GDNATIVE_EXT_VIDEODECODER:
238  hgdn_videodecoder_api = (const godot_gdnative_ext_videodecoder_api_struct *) hgdn_core_api->extensions[i];
239  break;
