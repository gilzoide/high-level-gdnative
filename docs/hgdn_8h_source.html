267 // Init and terminate
268 void hgdn_gdnative_init(const godot_gdnative_init_options *options) {
269  hgdn_core_api = options->api_struct;
270  for (const godot_gdnative_api_struct *ext = hgdn_core_api->next; ext; ext = ext->next) {
271  if (ext->version.major == 1 && ext->version.minor == 1) {
272  hgdn_core_1_1_api = (const godot_gdnative_core_1_1_api_struct *) ext;
273  } else if (ext->version.major == 1 && ext->version.minor == 2) {
274  hgdn_core_1_2_api = (const godot_gdnative_core_1_2_api_struct *) ext;
275  }
276  }
277 
278  for (unsigned int i = 0; i < hgdn_core_api->num_extensions; i++) {
279  switch(hgdn_core_api->extensions[i]->type) {
280  case GDNATIVE_EXT_NATIVESCRIPT:
281  hgdn_nativescript_api = (const godot_gdnative_ext_nativescript_api_struct *) hgdn_core_api->extensions[i];
282  for (const godot_gdnative_api_struct *ext = hgdn_nativescript_api->next; ext; ext = ext->next) {
283  if (ext->version.major == 1 && ext->version.minor == 1) {
284  hgdn_nativescript_1_1_api = (const godot_gdnative_ext_nativescript_1_1_api_struct *) ext;
285  }
286  }
287  break;
288 
289  case GDNATIVE_EXT_PLUGINSCRIPT:
290  hgdn_pluginscript_api = (const godot_gdnative_ext_pluginscript_api_struct *) hgdn_core_api->extensions[i];
291  break;
292 
293  case GDNATIVE_EXT_ANDROID:
294  hgdn_android_api = (const godot_gdnative_ext_android_api_struct *) hgdn_core_api->extensions[i];
295  break;
296 
297  case GDNATIVE_EXT_ARVR:
298  hgdn_arvr_api = (const godot_gdnative_ext_arvr_api_struct *) hgdn_core_api->extensions[i];
299  for (const godot_gdnative_api_struct *ext = hgdn_arvr_api->next; ext; ext = ext->next) {
300  if (ext->version.major == 1 && ext->version.minor == 2) {
301  hgdn_arvr_1_2_api = (const godot_gdnative_ext_arvr_1_2_api_struct *) ext;
302  }
303  }
304  break;
305 
306  case GDNATIVE_EXT_VIDEODECODER:
307  hgdn_videodecoder_api = (const godot_gdnative_ext_videodecoder_api_struct *) hgdn_core_api->extensions[i];
308  break;
309 
310  case GDNATIVE_EXT_NET:
311  hgdn_net_api = (const godot_gdnative_ext_net_api_struct *) hgdn_core_api->extensions[i];
312  for (const godot_gdnative_api_struct *ext = hgdn_net_api->next; ext; ext = ext->next) {
313  if (ext->version.major == 3 && ext->version.minor == 2) {
314  hgdn_net_3_2_api = (const godot_gdnative_ext_net_3_2_api_struct *) ext;
315  }
316  }
317  break;
318 
319  default:
320  break;
321  }
322  }
