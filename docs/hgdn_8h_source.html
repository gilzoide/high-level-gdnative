617  }

620  switch(hgdn_core_api->extensions[i]->type) {
621  case GDNATIVE_EXT_NATIVESCRIPT:

623  for (const godot_gdnative_api_struct *ext = hgdn_nativescript_api->next; ext; ext = ext->next) {

627  }

629

631  hgdn_pluginscript_api = (const godot_gdnative_ext_pluginscript_api_struct *) hgdn_core_api->extensions[i];

636  break;
637
